레퍼런스 관리

레퍼런스 페이지에 표시되는 프로젝트·캠페인 카드를 등록·수정합니다.

로그인: —

⚠️ Supabase 'references' 테이블이 아직 없어요

호성님 Supabase 콘솔(SQL Editor)에서 아래 SQL을 한 번만 실행하시면 됩니다. 그 후 새로고침하면 폼이 활성화됩니다.

create table if not exists public.references (
  id uuid primary key default gen_random_uuid(),
  category text not null check (category in ('sns','mgt','live','perf','content')),
  title text not null,
  client text,
  result_text text,
  tags text,
  image_url text,
  is_active boolean default true,
  display_order int default 100,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

alter table public.references enable row level security;

create policy "public read references"
  on public.references for select using (true);

create policy "auth all references"
  on public.references for all
  using (auth.role() = 'authenticated')
  with check (auth.role() = 'authenticated');
